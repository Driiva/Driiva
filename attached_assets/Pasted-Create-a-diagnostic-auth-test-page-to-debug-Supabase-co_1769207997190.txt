Create a diagnostic auth test page to debug Supabase connection:

1. Create file: client/src/pages/auth-test.tsx

2. Add this diagnostic component:

```tsx
import React, { useState } from 'react';
import { supabase } from '@/lib/supabase';

export default function AuthTest() {
  const [status, setStatus] = useState<string[]>([]);
  const [loading, setLoading] = useState(false);

  const addStatus = (msg: string) => {
    setStatus(prev => [...prev, `[${new Date().toLocaleTimeString()}] ${msg}`]);
    console.log(msg);
  };

  const testSupabaseConnection = async () => {
    setLoading(true);
    setStatus([]);
    
    try {
      // Test 1: Check if Supabase client exists
      addStatus('✓ Supabase client initialized');
      
      // Test 2: Try to signup
      addStatus('Testing signup...');
      const testEmail = `test-${Date.now()}@driiva.test`;
      const testPassword = 'TestPass123!';
      
      const { data: signupData, error: signupError } = await supabase.auth.signUp({
        email: testEmail,
        password: testPassword,
      });
      
      if (signupError) {
        addStatus(`✗ Signup failed: ${signupError.message}`);
      } else {
        addStatus(`✓ Signup success! User ID: ${signupData.user?.id}`);
        addStatus(`✓ Session created: ${signupData.session ? 'Yes' : 'No'}`);
        
        // Test 3: Check session
        const { data: sessionData } = await supabase.auth.getSession();
        addStatus(`✓ Current session: ${sessionData.session?.user?.email || 'None'}`);
        
        // Test 4: Try to insert profile
        addStatus('Testing profile insert...');
        const { data: profileData, error: profileError } = await supabase
          .from('profiles')
          .insert({
            id: signupData.user?.id,
            email: testEmail,
            first_name: 'Test',
            last_name: 'User',
            premium_amount: 1500.00
          })
          .select()
          .single();
        
        if (profileError) {
          addStatus(`✗ Profile insert failed: ${profileError.message}`);
          addStatus(`✗ Error code: ${profileError.code}`);
          addStatus(`✗ Error details: ${profileError.details}`);
        } else {
          addStatus(`✓ Profile created successfully!`);
        }
        
        // Test 5: Sign out
        await supabase.auth.signOut();
        addStatus('✓ Signed out');
      }
      
    } catch (err: any) {
      addStatus(`✗ Unexpected error: ${err.message}`);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-900 via-blue-900 to-purple-900 p-8">
      <div className="max-w-2xl mx-auto">
        <h1 className="text-3xl font-bold text-white mb-6">Supabase Auth Diagnostics</h1>
        
        <button
          onClick={testSupabaseConnection}
          disabled={loading}
          className="px-6 py-3 bg-[#06B6D4] hover:bg-[#0891B2] text-white rounded-lg font-medium mb-6 disabled:opacity-50"
        >
          {loading ? 'Testing...' : 'Run Auth Test'}
        </button>
        
        <div className="bg-gray-800/50 rounded-lg p-6 font-mono text-sm">
          <h2 className="text-white font-bold mb-4">Test Results:</h2>
          {status.length === 0 ? (
            <p className="text-gray-400">Click "Run Auth Test" to start diagnostics</p>
          ) : (
            <div className="space-y-2">
              {status.map((line, i) => (
                <div key={i} className={
                  line.includes('✓') ? 'text-green-400' : 
                  line.includes('✗') ? 'text-red-400' : 
                  'text-gray-300'
                }>
                  {line}
                </div>
              ))}
            </div>
          )}
        </div>
        
        <div className="mt-6 bg-yellow-900/20 border border-yellow-500/30 rounded-lg p-4">
          <p className="text-yellow-400 text-sm">
            <strong>What to look for:</strong><br/>
            - If signup fails: Check Supabase dashboard → Authentication settings<br/>
            - If profile insert fails with "permission denied": RLS policies need adjustment<br/>
            - If profile insert fails with "column does not exist": Table schema mismatch<br/>
            - If session is None: Auth not persisting to localStorage
          </p>
        </div>
      </div>
    </div>
  );
}
