Fix authentication routing so Home button goes to dashboard when logged in, and only Sign Out logs users out.

**Current Issue:**
- Home button takes user back to welcome page (wrong)
- No proper auth state management

**Correct Behavior:**
- Home button → /dashboard (when authenticated)
- Sign Out button → clears auth, redirects to welcome page
- Bottom nav Home icon → /dashboard (not welcome page)

---

**1. Create Auth Context for State Management:**

Create a new file: `client/src/context/AuthContext.tsx`

```typescript
import { createContext, useContext, useState, useEffect, ReactNode } from 'react'
import { useNavigate } from 'react-router-dom'

interface AuthContextType {
  isAuthenticated: boolean
  user: User | null
  login: (userData: User) => void
  logout: () => void
}

interface User {
  id: string
  name: string
  email: string
  // Add other user fields as needed
}

const AuthContext = createContext<AuthContextType | undefined>(undefined)

export function AuthProvider({ children }: { children: ReactNode }) {
  const [isAuthenticated, setIsAuthenticated] = useState<boolean>(false)
  const [user, setUser] = useState<User | null>(null)
  const navigate = useNavigate()

  // Check for existing session on mount
  useEffect(() => {
    const storedUser = localStorage.getItem('driiva_user')
    if (storedUser) {
      setUser(JSON.parse(storedUser))
      setIsAuthenticated(true)
    }
  }, [])

  const login = (userData: User) => {
    setUser(userData)
    setIsAuthenticated(true)
    localStorage.setItem('driiva_user', JSON.stringify(userData))
    navigate('/dashboard')
  }

  const logout = () => {
    setUser(null)
    setIsAuthenticated(false)
    localStorage.removeItem('driiva_user')
    navigate('/')
  }

  return (
    <AuthContext.Provider value={{ isAuthenticated, user, login, logout }}>
      {children}
    </AuthContext.Provider>
  )
}

export function useAuth() {
  const context = useContext(AuthContext)
  if (!context) {
    throw new Error('useAuth must be used within AuthProvider')
  }
  return context
}
